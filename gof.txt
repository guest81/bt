    local function shootAtTarget()
        local target = ReignMatcha.BulletRedirection.Internal.LockedTarget
        local targetChar = target and target.Character
        local targetHead = targetChar and targetChar:FindFirstChild("Head")
        if not targetHead then return end
        local argument1 = "Shoot"
        local argument2 = {
            [1]={}, [2]={},
            [3] = LocalPlayer.Character and LocalPlayer.Character.HumanoidRootPart and LocalPlayer.Character.HumanoidRootPart.Position or Vector3.zero,
            [4] = LocalPlayer.Character and LocalPlayer.Character.HumanoidRootPart and LocalPlayer.Character.HumanoidRootPart.Position or Vector3.zero,
            [5] = workspace:GetServerTimeNow()
        }
        for i=1,5 do
            table.insert(argument2[1], {Instance=targetHead, Normal=targetHead.Position, Position=targetHead.Position})
            table.insert(argument2[2], {thePart=targetHead, theOffset=CFrame.new(0,0,0)})
        end
        pcall(function() ReplicatedStorage.MainEvent:FireServer(argument1, argument2) end)
    end

    -- Kill Aura
    local Script = {
        Functions = {},
        Connections = {}
    }

    Script.Functions.GetClosestPlayer = function()
        if not Table.Aiming.SilentConfig.Kill_Aura then return end
        local closest, shortest = nil, Table.Aiming.SilentConfig.Kill_AuraRange
        for _, target in pairs(Players:GetPlayers()) do
            if target ~= LocalPlayer and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
                local dist = (target.Character.HumanoidRootPart.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
                if dist < shortest then closest=target; shortest=dist end
            end
        end
        return closest
    end

    Script.Functions.KillAura = function()
        local closest = Script.Functions.GetClosestPlayer()
        if not closest then return end
        local tool = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Tool")
        if closest and tool and not closest.Character:FindFirstChildOfClass("ForceField") then
            UseDatArgBoiV2(closest)
        end
    end

    RunService.Heartbeat:Connect(function()
        if Table.Aiming.SilentConfig.Kill_Aura then Script.Functions.KillAura() end
    end)

    -- Silent Aim Logic
    RunService.Heartbeat:Connect(function()
        local target = ReignMatcha.BulletRedirection.Internal.LockedTarget
        if not target then return end
        local target_char = target.Character
        if not target_char then return end
        local target_hrp  = target_char:FindFirstChild("HumanoidRootPart")
        local target_head = target_char:FindFirstChild("Head")
        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp or not target_hrp then return end
        local dist      = (hrp.Position - target_hrp.Position).Magnitude
        local direction = (target_hrp.Position - hrp.Position).Unit * dist
        local ray_params = RaycastParams.new()
        ray_params.FilterType = Enum.RaycastFilterType.Blacklist
        ray_params.FilterDescendantsInstances = {char, target_char}
        ray_params.IgnoreWater = true
        local is_visible = not workspace:Raycast(hrp.Position, direction, ray_params)
        local in_range   = dist <= ReignMatcha.BulletRedirection.Settings.MagicBulletRange
        local place_id   = game.PlaceId
        local valid_target = in_range and (place_id==2788229376 or is_visible)
        if place_id==2788229376 then
            if valid_target then UseDatArgBoi() end
        else
            if valid_target then UseDatArgBoi() end
        end
        if mouseDown and manualclickfh then
            if not usechecksonmanual or valid_target then UseDatArgBoi() end
        end
    end)

    local function GetAimbotTarget()
        local cam = workspace.CurrentCamera
        local closest, minDist = nil, ReignMatcha.Aimbot.FOV
        for _, p in ipairs(Players:GetPlayers()) do
            if p~=LocalPlayer and p.Character then
                local part = p.Character:FindFirstChild(ReignMatcha.Aimbot.AimPart) or p.Character:FindFirstChild("HumanoidRootPart")
                local hum  = p.Character:FindFirstChild("Humanoid")
                if part and hum and hum.Health>0 then
                    local vec, onScreen = cam:WorldToViewportPoint(part.Position)
                    if onScreen then
                        local dist2D = (Vector2.new(vec.X,vec.Y) - cam.ViewportSize/2).Magnitude
                        if dist2D < minDist then minDist=dist2D; closest=p end
                    end
                end
            end
        end
        return closest
    end

    RunService.RenderStepped:Connect(function()
        if not ReignMatcha.Aimbot.Enabled then return end
        local target = ReignMatcha.Aimbot.LockedTarget or GetAimbotTarget()
        if not target or not target.Character then return end
        local part = target.Character:FindFirstChild(ReignMatcha.Aimbot.AimPart) or target.Character:FindFirstChild("HumanoidRootPart")
        if not part then return end
        local vel       = part.AssemblyLinearVelocity or Vector3.zero
        local predicted = part.Position + vel * ReignMatcha.Aimbot.Prediction
        predicted = predicted + Vector3.new(0, ReignMatcha.Aimbot.Yaxis/100, 0)
        workspace.CurrentCamera.CFrame = workspace.CurrentCamera.CFrame:Lerp(
            CFrame.new(workspace.CurrentCamera.CFrame.Position, predicted),
            ReignMatcha.Aimbot.Smoothness
        )
    end)

    local function render_esp_for_player(plr)
        if ESPHolder:FindFirstChild(plr.Name) then ESPHolder[plr.Name]:Destroy() end

        local container = Create("Folder", { Parent=ESPHolder, Name=plr.Name })

        local Name      = Create("TextLabel", { Parent=ESPHolder, BackgroundTransparency=1, TextColor3=Color3.new(1,1,1), Font=Enum.Font.Code, TextSize=ESP.FontSize, TextStrokeTransparency=0, RichText=true, Size=UDim2.new(0,100,0,20) })
        local Distance  = Create("TextLabel", { Parent=ESPHolder, BackgroundTransparency=1, TextColor3=Color3.new(1,1,1), Font=Enum.Font.Code, TextSize=ESP.FontSize, TextStrokeTransparency=0, Size=UDim2.new(0,100,0,20) })
        local Box       = Create("Frame",     { Parent=ESPHolder, BackgroundColor3=Color3.new(0,0,0), BackgroundTransparency=0.75, BorderSizePixel=0 })

        local Healthbar = Create("Frame",     { Parent=ESPHolder, BackgroundColor3=Color3.new(1,1,1), BackgroundTransparency=0 })
        local BehindHB  = Create("Frame",     { Parent=ESPHolder, ZIndex=-1, BackgroundColor3=Color3.new(0,0,0), BackgroundTransparency=0 })
        local Chams     = Create("Highlight", { Parent=ESPHolder, FillTransparency=1, OutlineTransparency=0, OutlineColor=Color3.fromRGB(119,120,255), DepthMode=Enum.HighlightDepthMode.AlwaysOnTop })

        local conn
        conn = RunService.RenderStepped:Connect(function()
            if not plr or not plr.Parent then
                Name.Visible=false; Box.Visible=false; Distance.Visible=false
                Healthbar.Visible=false; BehindHB.Visible=false; Chams.Enabled=false
                conn:Disconnect(); return
            end

            local char = plr.Character
            if not char then Name.Visible=false; Box.Visible=false; Healthbar.Visible=false; BehindHB.Visible=false; Chams.Enabled=false; return end

            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChild("Humanoid")
            if not hrp or not hum then Name.Visible=false; Box.Visible=false; Healthbar.Visible=false; BehindHB.Visible=false; Chams.Enabled=false; return end

            if not ESP.Enabled then Name.Visible=false; Box.Visible=false; Healthbar.Visible=false; BehindHB.Visible=false; Chams.Enabled=false; return end

            local cam = workspace.CurrentCamera
            local vec, onScreen = cam:WorldToViewportPoint(hrp.Position)
            local Dist = (hrp.Position - cam.CFrame.Position).Magnitude

            if Dist > ESP.MaxDistance then Name.Visible=false; Box.Visible=false; Healthbar.Visible=false; BehindHB.Visible=false; Chams.Enabled=false; return end

            if onScreen then
                local topY    = cam:WorldToViewportPoint(hrp.Position + Vector3.new(0,3,0)).Y
                local bottomY = cam:WorldToViewportPoint(hrp.Position - Vector3.new(0,3,0)).Y
                local h = math.abs(bottomY - topY)
                local w = h / 2

                -- Box
                Box.Visible  = ESP.Drawing.Boxes.Full.Enabled
                Box.Position = UDim2.new(0, vec.X-w/2, 0, vec.Y-h/2)
                Box.Size     = UDim2.new(0, w, 0, h)
                Box.BackgroundTransparency = ESP.Drawing.Boxes.Filled.Enabled and ESP.Drawing.Boxes.Filled.Transparency or 1

                -- Name
                Name.Visible   = ESP.Drawing.Names.Enabled
                Name.Text      = ESP.Drawing.Distances.Position=="Text"
                    and string.format("%s [%d]", plr.Name, math.floor(Dist))
                    or  plr.Name
                Name.Position  = UDim2.new(0, vec.X, 0, vec.Y-h/2-9)
                Name.TextColor3 = ESP.Drawing.Names.RGB

                -- Distance
                Distance.Visible   = ESP.Drawing.Distances.Enabled and ESP.Drawing.Distances.Position=="Bottom"
                Distance.Text      = string.format("%d m", math.floor(Dist))
                Distance.Position  = UDim2.new(0, vec.X, 0, vec.Y+h/2+7)

                -- Health bar
                local healthPct = hum.Health / math.max(hum.MaxHealth,1)
                Healthbar.Visible  = ESP.Drawing.Healthbar.Enabled
                Healthbar.Position = UDim2.new(0, vec.X-w/2-6, 0, vec.Y-h/2+h*(1-healthPct))
                Healthbar.Size     = UDim2.new(0, ESP.Drawing.Healthbar.Width, 0, h*healthPct)
                BehindHB.Visible   = ESP.Drawing.Healthbar.Enabled
                BehindHB.Position  = UDim2.new(0, vec.X-w/2-6, 0, vec.Y-h/2)
                BehindHB.Size      = UDim2.new(0, ESP.Drawing.Healthbar.Width, 0, h)

                -- Chams
                if ESP.Drawing.Chams.Enabled then
                    Chams.Adornee  = char
                    Chams.Enabled  = true
                    Chams.FillColor    = ESP.Drawing.Chams.FillRGB
                    Chams.OutlineColor = ESP.Drawing.Chams.OutlineRGB
                    Chams.FillTransparency = ESP.Drawing.Chams.Fill_Transparency/100
                    Chams.OutlineTransparency = ESP.Drawing.Chams.Outline_Transparency/100
                else
                    Chams.Enabled = false
                end
            else
                Name.Visible=false; Box.Visible=false; Distance.Visible=false
                Healthbar.Visible=false; BehindHB.Visible=false; Chams.Enabled=false
            end
        end)
    end
